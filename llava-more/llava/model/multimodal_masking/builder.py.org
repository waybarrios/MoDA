import torch
import torch.nn as nn
import torch.nn.functional as F

class EncoderMask(nn.Module):
    """
    Define a learnable function to generate a sparse mask based on input features.
    Uses SiLU activations and embedding layers without bias.
    """
    def __init__(self, num_tokens=576, embedding_dim=4096, hidden_dim=1024, l1_lambda=1e-5):
        super().__init__()

        # Embedding layers without bias
        self.embedding1 = nn.Linear(embedding_dim, hidden_dim, bias=False)
        self.embedding2 = nn.Linear(hidden_dim, embedding_dim, bias=False)
        self.activation = nn.SiLU()  # Using nn.SiLU instead of F.silu
        
        self.l1_lambda = l1_lambda  # Weight for L1 regularization

        # Initialize weights using Xavier initialization
        nn.init.xavier_uniform_(self.embedding1.weight)
        nn.init.xavier_uniform_(self.embedding2.weight)

    def forward(self, x):
        """
        x: Tensor of shape (B, N, D)
        Computes a sparse mask using a learned function and applies it to x.
        """
        # Compute the mask function
        mask = self.embedding1(x)  # (B, N, hidden_dim)
        mask = self.activation(mask)  # Apply SiLU activation using nn.SiLU()
        mask = self.embedding2(mask)  # (B, N, D)
        mask = torch.sigmoid(mask)  # (B, N, D), values in (0,1)
        
        return x * mask  # Apply mask

    def l1_loss(self):
        """ Computes L1 loss to encourage sparsity in the mask. """
        return self.l1_lambda * F.l1_loss(self.embedding2.weight, torch.zeros_like(self.embedding2.weight), reduction="sum")

# Function to build the vision masking model
def build_vision_masking(num_tokens=576, embedding_dim=4096, hidden_dim=1024, l1_lambda=1e-5):
    return EncoderMask(num_tokens=num_tokens, embedding_dim=embedding_dim, hidden_dim=hidden_dim, l1_lambda=l1_lambda)
